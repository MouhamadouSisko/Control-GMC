import tkinter as tk

def fahrenheit_to_celsius():
    """Converts the temperature in Fahrenheit to Celsius"""
    fahrenheit = ent_temperature.get()
    celsius = round((float(fahrenheit) - 32) * 5/9,2)
    lbl_result.config(text=f"{celsius:.2f} \N{DEGREE CELSIUS}")

# Créer la fenêtre principale
window = tk.Tk()
window.title("Convertisseur de température")
window.resizable(width=False, height=False)

# Créer un cadre pour les widgets d'entrée
frm_entry = tk.Frame(master=window)
ent_temperature = tk.Entry(master=frm_entry, width=10)
lbl_temp = tk.Label(master=frm_entry, text="\N{DEGREE FAHRENHEIT}")

# Utilisez le gestionnaire de géométrie de grille pour organiser les widgets
ent_temperature.grid(row=0, column=0, sticky="e")
lbl_temp.grid(row=0, column=1, sticky="w")
frm_entry.grid(row=0, column=0, padx=10)

# Créer un widget bouton pour lancer la conversion
btn_convert = tk.Button(
    master=window,
    text="\N{RIGHTWARDS BLACK ARROW}",
    command=fahrenheit_to_celsius
)
# Créer une étiquette pour afficher le résultat de la conversion
lbl_result = tk.Label(master=window, text="\N{DEGREE CELSIUS}")

# Utilisez le gestionnaire de géométrie de grille pour organiser les widgets
btn_convert.grid(row=0, column=1, pady=10)
lbl_result.grid(row=0, column=2, padx=10)

# Lancer l'application
window.mainloop()
